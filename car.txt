<script>
    (function() {
        const car = document.getElementById('movingCar');
        const track = document.getElementById('raceTrack');
        const replayBtn = document.getElementById('replayBtn');
        
        // 重启动画函数：完美重启CSS动画（时长会从CSS中读取，已改为4秒）
        function restartCarAnimation() {
            if (!car) return;
            // 移除当前动画样式
            car.style.animation = 'none';
            // 强制重绘，确保动画重新开始
            void car.offsetHeight;
            // 重新应用动画（时长已在CSS中定义为4s）
            car.style.animation = 'driveDiagonal 4s cubic-bezier(0.25, 0.46, 0.45, 0.94) forwards';
        }
        
        // 绑定重播按钮事件
        if (replayBtn) {
            replayBtn.addEventListener('click', (e) => {
                e.preventDefault();
                restartCarAnimation();
                // 按钮点击微反馈
                replayBtn.style.transform = 'scale(0.96)';
                setTimeout(() => {
                    if (replayBtn) replayBtn.style.transform = '';
                }, 120);
            });
        }
        
        // 双击赛道也可以重新播放动画
        if (track) {
            track.addEventListener('dblclick', (e) => {
                e.stopPropagation();
                restartCarAnimation();
            });
        }
        
        // 页面加载时确保汽车从左上角开始，并应用慢速动画
        window.addEventListener('load', () => {
            if (car) {
                car.style.animation = 'none';
                void car.offsetHeight;
                car.style.animation = 'driveDiagonal 4s cubic-bezier(0.25, 0.46, 0.45, 0.94) forwards';
            }
        });
    })();
</script>
